Simple Form to Google Sheets

Easy Setup Instructions

  1. Create a new Google Sheet
  2. Click Extensions > Apps Script
  3. Delete any code and paste the code from the box below
  4. Click Deploy > New deployment
  5. Select Web app as the type
  6. Set Execute as to "Me" and Who has access to "Anyone"
  7. Click Deploy and copy the provided URL
  8. Paste the URL in the field below

Google Apps Script Code

Copy and paste this code into your Google Apps Script editor:

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    
    // Get parameters from the request
    var params = e.parameter;
    
    // Prepare the row data
    var row = [
      new Date(), // Timestamp
      params.name || '',
      params.email || '',
      params.service || '',
      params.message || ''
    ];
    
    // Append the row to the sheet
    sheet.appendRow(row);
    
    // Return a success response
    return ContentService.createTextOutput("Success: Data saved to sheet");
    
  } catch (error) {
    // Return an error response
    return ContentService.createTextOutput("Error: " + error.toString());
  }
}
